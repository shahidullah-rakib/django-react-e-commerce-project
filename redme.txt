Backend
1. python -m venv env
2. env\Scripts\activate.bat
3. pip install djangorestframework djangorestframework-simplejwt
4. pip install django
5. django-admin startproject backend
6. pip install mysqlclient
7. pip install django-cors-headers
8. python manage.py create_groups
